/*Дано   катети   прямокутного   трикутника.   Знайти   його гіпотенузу  і  площу */

//Обчислення гіпотенузи та площі//

#include <stdio.h>
#include <math.h>
#include <stdlib.h>
#include <conio.h>
int main() 
{	
   system("chcp 1251 \n");
 
   printf("\n Введіть вихідні дані: \n");
    
   //Дійсні числа//
   float a, b;
   int p;
	  
   //Введення даних//
   printf("\n Катет 1 -> ");
	scanf("%f", &a);
		
   printf("\n Катет 2 -> ");
	scanf("%f", &b);
	 
   //Розрахунок гіпотенузи та площі трикутника//
   printf("\n Гіпотенуза: %.1f \n", sqrt(a*a+b*b));
   p = (a+b+sqrt(a*a+b*b))/2;
   printf("\n Площа: %.1f \n", sqrt(p*(p-a)*(p-b)*(p-sqrt(a*a+b*b))));
 	
   getch();
   return 1;
}
