/* Написати   програму   обчислення вартості   покупки   з урахуванням  знижки.  
Знижка в 3%  надається  в  тому випадку, якщо сума покупки більше  500 грн., 
в 5% - якщо сума більше 1000 грн. */


//Обчислення  вартості  покупки  з  урахуванням знижки//

#include <stdio.h>
#include <conio.h>

int main ()
{  
  int discount, summ, tsumm;
  
  // Ввід даних //
  printf ("\n Input your summ: "); 
  	scanf ("%d", &summ);
  
  // Алгоритм обчислення вартості   покупки   з урахуванням  знижки //
  if ( summ < 500 )
  	printf ("\n You haven't a discount! ");
  else 
       {
  	  if ( summ > 1000 ) { 
	  printf ("\n You discount: 5%%");
	  discount = 5; 
	  tsumm = summ - (discount * summ)/100; }
	  
	  else { 
	  printf ("\n You discount: 3%%");
	  discount = 3; 
	  tsumm = summ - (discount * summ)/100; }

     printf ("\n The total summ = %3.2d", tsumm);
   }	
    
 getch();
 return 0;
}
